165 	private class BtnAtrsActionListener implements ActionListener {
166 		public void actionPerformed(ActionEvent e) {
167 			Cliente c= new Cliente(lblLblnick.getText());
168 			GestorCliente gc = new GestorCliente();
169 			try {
170 				c=gc.obtenerCliente(c);
171 			} catch (Exception e2) {
172 				e2.printStackTrace();
173 			}
174 			pwdPass.setText(c.getPass());
175 			txtEmail.setText(c.getEmail());
176 			txtApellidos.setText(c.getApellidos());
177 			txtNombre.setText(c.getNombre());
178 		}
179 	}
180 	private class BtnActualizarMisDatosActionListener implements ActionListener {
181 		public void actionPerformed(ActionEvent e) {
182 			GestorCliente gc = new GestorCliente();
183 			String suscripcion;
184 			if (chckbxActualizarAPremium.isSelected()) {
185 				suscripcion="Premium";
186 			} else {
187 				suscripcion="Normal";
188 			}
189 			Cliente client = new Cliente(lblLblnick.getText(), pwdPass.getText(), txtEmail.getText(), txtNombre.getText(), txtApellidos.getText(), suscripcion, "Usuario");
190 			try {
191 				gc.actualizarCliente(client);
192 				JOptionPane.showMessageDialog(null, "Tu cuenta se ha actualizado con éxito");
193 			} catch (Exception e2) {
194 				e2.printStackTrace();
195 			}
196 			
197 		}
198 	}
199 	private class BtnEliminarMiCuentaActionListener implements ActionListener {
200 		public void actionPerformed(ActionEvent e) {
201 			int confirmado = JOptionPane.showConfirmDialog(null, "¿Estás seguro de que quieres eliminar tu cuenta? Esta acción no se puede deshacer."
202 					,"Ventana de confirmación",JOptionPane.YES_NO_OPTION);
203 			if (confirmado==JOptionPane.YES_OPTION) {
204 				GestorCliente gc = new GestorCliente();
205 				Cliente client = new Cliente(lblLblnick.getText());
206 				try {
207 
208 					gc.eliminarCliente(client);
209 					JOptionPane.showMessageDialog(null, "Tu cuenta se ha eliminado con éxito. Saliendo...");
210 					System.exit(0);
211 
212 				} catch (Exception e2) {
213 					e2.printStackTrace();
214 					
215 				}
216 			}
217 		}
218 	}
