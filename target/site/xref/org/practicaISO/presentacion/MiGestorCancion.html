157 	private class BtnEliminarCancinActionListener implements ActionListener {
158 		public void actionPerformed(ActionEvent e) {
159 			int confirmado = JOptionPane.showConfirmDialog(null,
160 					"¿Estás seguro de que quieres eliminar esta canción? Esta acción no se puede deshacer.",
161 					"Ventana de confirmación", JOptionPane.YES_NO_OPTION);
162 			if (confirmado == JOptionPane.YES_OPTION) {
163 				Cancion canc = new Cancion(txtTitulo.getText());
164 				try {
165 					canc = canc.obtenerCancionTitulo();
166 					canc.eliminarCancion();
167 				} catch (Exception e2) {
168 					e2.printStackTrace();
169 				}
170 
171 			}
172 		}
173 	}
174 
175 	private class BtnActualizarCancinActionListener implements ActionListener {
176 		public void actionPerformed(ActionEvent e) {
177 			int confirmado = JOptionPane.showConfirmDialog(null,
178 					"¿Estás seguro de que quieres actualizar esta canción?", "Ventana de confirmación",
179 					JOptionPane.YES_NO_OPTION);
180 			if (confirmado == JOptionPane.YES_OPTION) {
181 				Cancion canc = new Cancion(txtTitulo.getText(), Integer.parseInt(lblId.getText()), txtArtista.getText(),
182 						Integer.parseInt(txtAlbum.getText()), Float.parseFloat(txtPrecio.getText()));
183 
184 				try {
185 					canc.actualizarCancion();
186 					JOptionPane.showMessageDialog(null, "Canción actualizada con éxito");
187 				} catch (Exception e2) {
188 					e2.printStackTrace();
189 				}
190 			}
191 			dispose();
192 		}
