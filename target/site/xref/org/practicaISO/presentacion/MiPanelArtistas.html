119 	public void actualizarModelo(Cliente client) {
120 		GestorCliente gc = new GestorCliente();
121 		GestorCancion gcan= new GestorCancion();
122 		ArrayList<Cancion> ac=null;
123 		DefaultListModel<String> modelo = new DefaultListModel<String>();
124 		DefaultListModel<String> modelolimpio = new DefaultListModel<String>();
125 		try {
126 			ac=gc.obtenerCanciones(client);		
127 		} catch (Exception e) {
128 			System.out.println("Error al obtener canciones del usuario");
129 		}
130 		if(ac!=null) {
131 			for (Cancion canc : ac) {
132 				try {
133 					canc=gcan.obtenerCancionId(canc);
134 					modelo.addElement(canc.getAutor());
135 
136 				} catch (Exception e) {
137 					e.printStackTrace();
138 				}
139 			
140 			}
141 			//para no repetir albumes
142 			for (int ii=0;ii<modelo.size();ii++) {
143 				for (int jj=0;jj<modelo.size();jj++) {
144 					if(!modelo.get(ii).equals(modelo.get(jj))) {
145 						if(!modelolimpio.contains(modelo.get(jj))) {
146 							modelolimpio.addElement(modelo.get(jj));
147 						}
148 					}
149 				}
150 			}
151 		}else {
152 			JOptionPane.showMessageDialog(this, "No hay artistas que mostrar");
153 		}
154 		list.setModel(modelolimpio);
155 	}
